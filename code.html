                  // get the values from the UI (controller function)
                  function getValues() {
                    // get values from the page
                    let startValue = document.getElementById('startValue').value;
                    let endValue = document.getElementById('endValue').value;

                    // parse into Integers
                    startValue = parseInt(startValue);
                    endValue = parseInt(endValue);

                    if(Number.isInteger(startValue) && Number.isInteger(endValue)) {
                      let numbers = generateNumbers(startValue, endValue);
                      // call displayNumbers
                      displayNumbers(numbers);
                    } else {
                      alert("Error! You must enter integers.");
                    }

                  }

                  // generate numbers from the start value to the end value (logic function)
                  function generateNumbers(start, end) {

                    let numbers = [];

                    for(let i = start; i <= end; i++){
                      numbers.push(i);
                    }

                    return numbers;
                  }

                  // display the numbers and mark the even numbers bold (display or view function)
                  function displayNumbers(numbers) {

                    let templateRows = "";

                    for (let i = 0; i < numbers.length; i++) {
                      let number = numbers[i];
                      numbers[i] % 2 === 0 ? templateRows += `<tr><td><b>${number}</b></td></tr>` : templateRows += `<tr><td>${number}</td></tr>`;
                    }

                    document.getElementById('results').innerHTML = templateRows;
                  }

The code is structured using three functions.

Get Values

getValues is a function that retrieves the values of the inputs from the Enter Values section on The App page. It validates the input, ensuring that the values entered into the inputs are in fact numbers. Once the function verifies this, it calls another function called generateNumbers, passing as arguments the start value and the end value from the inputs. This returns an array containing those numbers (along with every number in between), which gets passed to displayNumbers, which as the name suggests handles displaying the numbers on the screen.

Generate Numbers

generateNumbers is the second function used in this coding challenge. A for loop is used, with the starting value and ending value as the minimum and maximum values, to push each number to an array called numbers. This array is returned.

Display Numbers

displayNumbers is our third and final function used in this coding challenge. An empty string, called, templateRows, will be used to hold the same HTML one would write to display the numbers on The App page. Another for loop is used to process each number in the numbers array from generateNumbers. Within this for loop is located a statement using the ternary operator, which is essentially an alternative to an if-else statement; this line of code bolds the numbers that are even and does nothing to the numbers that are odd. The result of this statement is appended to templateRows and is added to the body of the table created to hold the results, which has an Id of 'results', on line 43.
